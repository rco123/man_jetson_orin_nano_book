OpenCV 색상 처리 예제

Jetson Nano에서 OpenCV를 사용하여 이미지 또는 비디오의 색상을 처리하는 예제입니다.

1. 색상 공간 변환

이미지의 색상 공간을 BGR에서 HSV로 변환하거나, 다른 색상 공간으로 변환하는 기본 예제입니다.

코드:

import cv2

def main():
    # 이미지를 읽어옵니다
    image = cv2.imread("sample.jpg")
    if image is None:
        print("이미지를 로드할 수 없습니다.")
        return

    # BGR에서 HSV로 변환
    hsv_image = cv2.cvtColor(image, cv2.COLOR_BGR2HSV)

    # BGR에서 GRAY로 변환
    gray_image = cv2.cvtColor(image, cv2.COLOR_BGR2GRAY)

    # 이미지 출력
    cv2.imshow("Original Image", image)
    cv2.imshow("HSV Image", hsv_image)
    cv2.imshow("Gray Image", gray_image)

    cv2.waitKey(0)
    cv2.destroyAllWindows()

if __name__ == "__main__":
    main()

2. 특정 색상 필터링

HSV 색상 공간에서 특정 색상(예: 빨간색, 파란색 등)을 추출합니다.

코드:

import cv2
import numpy as np

def main():
    # 이미지를 읽어옵니다
    image = cv2.imread("sample.jpg")
    if image is None:
        print("이미지를 로드할 수 없습니다.")
        return

    # BGR에서 HSV로 변환
    hsv = cv2.cvtColor(image, cv2.COLOR_BGR2HSV)

    # 빨간색 범위 정의
    lower_red = np.array([0, 120, 70])
    upper_red = np.array([10, 255, 255])

    # 빨간색에 해당하는 마스크 생성
    mask = cv2.inRange(hsv, lower_red, upper_red)

    # 원본 이미지와 마스크 적용
    result = cv2.bitwise_and(image, image, mask=mask)

    # 결과 출력
    cv2.imshow("Original Image", image)
    cv2.imshow("Red Color Filtered", result)

    cv2.waitKey(0)
    cv2.destroyAllWindows()

if __name__ == "__main__":
    main()

3. 이미지의 색상 히스토그램

이미지의 색상 분포를 히스토그램으로 계산합니다.

코드:

import cv2
import numpy as np
from matplotlib import pyplot as plt

def main():
    # 이미지를 읽어옵니다
    image = cv2.imread("sample.jpg")
    if image is None:
        print("이미지를 로드할 수 없습니다.")
        return

    # 각 채널별 히스토그램 계산
    colors = ('b', 'g', 'r')
    for i, color in enumerate(colors):
        hist = cv2.calcHist([image], [i], None, [256], [0, 256])
        plt.plot(hist, color=color)
        plt.xlim([0, 256])

    plt.title("Color Histogram")
    plt.xlabel("Pixel Intensity")
    plt.ylabel("Frequency")
    plt.show()

if __name__ == "__main__":
    main()

4. 색상 반전

이미지의 색상을 반전하는 예제입니다.

코드:

import cv2

def main():
    # 이미지를 읽어옵니다
    image = cv2.imread("sample.jpg")
    if image is None:
        print("이미지를 로드할 수 없습니다.")
        return

    # 색상 반전
    inverted_image = cv2.bitwise_not(image)

    # 결과 출력
    cv2.imshow("Original Image", image)
    cv2.imshow("Inverted Image", inverted_image)

    cv2.waitKey(0)
    cv2.destroyAllWindows()

if __name__ == "__main__":
    main()

5. 이미지 블러링 및 색상 변경

이미지를 블러링한 뒤 특정 색상만 강조합니다.

코드:

import cv2
import numpy as np

def main():
    # 이미지를 읽어옵니다
    image = cv2.imread("sample.jpg")
    if image is None:
        print("이미지를 로드할 수 없습니다.")
        return

    # 이미지 블러링
    blurred = cv2.GaussianBlur(image, (15, 15), 0)

    # HSV로 변환 후 특정 색상만 추출 (파란색 예제)
    hsv = cv2.cvtColor(blurred, cv2.COLOR_BGR2HSV)
    lower_blue = np.array([100, 150, 0])
    upper_blue = np.array([140, 255, 255])
    mask = cv2.inRange(hsv, lower_blue, upper_blue)

    # 원본 이미지에 색상 강조
    result = cv2.bitwise_and(image, image, mask=mask)

    # 결과 출력
    cv2.imshow("Blurred Image", blurred)
    cv2.imshow("Blue Color Enhanced", result)

    cv2.waitKey(0)
    cv2.destroyAllWindows()

if __name__ == "__main__":
    main()
